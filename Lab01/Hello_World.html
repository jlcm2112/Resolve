///-------------------------------------------------------------
/// Global Context ---------------------------------------------
///-------------------------------------------------------------

#include "RESOLVE_Foundation.h"


///-------------------------------------------------------------
/// Interface --------------------------------------------------
///-------------------------------------------------------------

program_body main ()
/*!
    ensures
	stdout = #stdout * "Hello world\n"
!*/
{
    object Character_OStream outs;
    
    outs.Open_External ("");
    outs << "Hello world\n";
    outs.Close_External ();
}
